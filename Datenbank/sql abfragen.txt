FROM person p

FROM person p  
JOIN geschlecht g on p.Geschlecht = g.ID
JOIN trainingszustand t on p.trainingszustand = t.ID 
WHERE p.Vorname = "Peter" AND p.Nachname = "Parker"

(SELECT id FROM aktivitaet a WHERE a.Art = "Intervallmethode")

2.6
SELECT p.id, p.Geburtsdatum, AVG(d.wert)
FROM person p
JOIN pulsdaten d ON p.ID = d.person
GROUP BY d.person;

2.7 ist das richtig?
SELECT  (220 - FLOOR(DATEDIFF(CURRENT_DATE(),p.Geburtsdatum)/365)) as "kritischer Bereich"
FROM person p
JOIN pulsdaten d ON p.ID = d.person
GROUP BY d.person
HAVING "kritsischer Bereich" = 12;

SELECT count(p.ID) as "Anzahl der Personen die trainiert haben" FROM person p
JOIN pulsdaten d ON p.ID = d.person
WHERE MONTH(d.erfassungszeit) = 2
AND YEAR(d.erfassungszeit) = 2022;
